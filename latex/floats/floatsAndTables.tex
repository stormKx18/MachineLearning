%New table
Table~\ref{tab:multi} used row and column span.

\begin{table}[htbp]
\caption{Spanning rows and columns}
\begin{center}
\begin{tabular}{| l | c | c |}
\hline
& \multicolumn{2}{c|}{Ranges} \\ % <----- Multi column
& X & Y \\
\hline
\multirow{3}{*}{Hot} & 7 & 9 \\  % <----- Multi row
& 5 & 8 \\
& 6 & 7 \\
\hline
\multirow{3}{*}{Cold} & 4 & 7 \\
& 2 & 9 \\
& 3 & 5 \\
\hline
\end{tabular}
\end{center}
\label{tab:multi}
\end{table}
